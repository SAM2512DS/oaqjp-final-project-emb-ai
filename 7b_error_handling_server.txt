"""
Flask server for Emotion Detection application.
Provides web interface for emotion analysis with error handling.
"""

from flask import Flask, render_template, request
from EmotionDetection import emotion_detector

app = Flask(__name__)

@app.route('/')
def index():
    """Render the main page."""
    return render_template('index.html')

@app.route('/emotionDetector')
def emotion_detector_route():
    """
    Process text input and return emotion analysis results.
    Handles blank input with appropriate error message.
    """
    text_to_analyze = request.args.get('textToAnalyze')
    
    if not text_to_analyze or text_to_analyze.strip() == "":
        return "Invalid text! Please try again!"
    
    result = emotion_detector(text_to_analyze)
    
    if result is None:
        return "Error processing the request. Please try again."
    
    if result.get('dominant_emotion') is None:
        return "Invalid text! Please try again!"
    
    response = (f"For the given statement, the system response is "
                f"'anger': {result['anger']}, "
                f"'disgust': {result['disgust']}, "
                f"'fear': {result['fear']}, "
                f"'joy': {result['joy']}, "
                f"'sadness': {result['sadness']}. "
                f"The dominant emotion is {result['dominant_emotion']}.")
    
    return response

if __name__ == '__main__':
    app.run(debug=True, host='0.0.0.0', port=5000)
